  group_by(nome_popular) %>%
  summarise(
    media_preco = round(mean(preco_r),1),
    media_rendimento = round(mean(rendimento_r),1),
    freq_ab = n()) %>%
  mutate(freq = round(100 * freq_ab/sum(freq_ab), 3)) %>%  
  arrange(desc(freq)) %>% 
  adorn_totals("row") %>%
  kable() %>% 
  kable_styling(bootstrap_options = c("striped", "hover", "responsive"),
